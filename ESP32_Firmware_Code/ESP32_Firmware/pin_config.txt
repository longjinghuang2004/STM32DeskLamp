# ==============================================================================
# 项目名称: AI 智能台灯 (ESP32-S3 大脑端)
# 硬件版本: V3.0 (基于 ESP32-S3-DevKitC-1)
# 日期: 2023-10-xx
# ==============================================================================

# ------------------------------------------------------------------------------
# [1] 系统与调试 (System & Debug)
# ------------------------------------------------------------------------------
# USB_D-            : GPIO 19  (保留，用于 USB 下载/JTAG)
# USB_D+            : GPIO 20  (保留，用于 USB 下载/JTAG)
# UART0_TX          : GPIO 43  (保留，用于 Log 查看)
# UART0_RX          : GPIO 44  (保留，用于 Log 查看)
# Onboard_RGB_LED   : GPIO 48  (板载 WS2812，用于状态指示)
# Boot_Button       : GPIO 0   (系统启动引脚，仅用于下载)
# JTAG              : GPIO 3   (保留，防止干扰调试)
# Strap             : GPIO 45  (电源电压配置，不建议使用)
# PSRAM/Flash       : GPIO 35,36,37 (严禁使用 - Octal PSRAM)

# ------------------------------------------------------------------------------
# [2] 视觉系统 (Display - 8080 Parallel Interface)
# 设备: ILI9486 (LCD) - 8位并行 + DMA
# 性能: 理论刷新率 > 50 FPS
# 位置: 左侧排针连续引脚
# ------------------------------------------------------------------------------
# LCD_D0            : GPIO 4   (数据总线 D0)
# LCD_D1            : GPIO 5   (数据总线 D1)
# LCD_D2            : GPIO 6   (数据总线 D2)
# LCD_D3            : GPIO 7   (数据总线 D3)
# LCD_D4            : GPIO 8   (数据总线 D4)
# LCD_D5            : GPIO 9   (数据总线 D5)
# LCD_D6            : GPIO 10  (数据总线 D6)
# LCD_D7            : GPIO 11  (数据总线 D7)
# LCD_WR            : GPIO 12  (写入触发 - 写时钟)
# LCD_CS            : GPIO 13  (片选 - 选中屏幕)
# LCD_DC            : GPIO 14  (数据/命令切换)
# LCD_RST           : GPIO 15  (硬件复位)
# LCD_BL            : GPIO 16  (背光控制 - PWM 调光，建议接 5V)

# ------------------------------------------------------------------------------
# [3] 神经连接 (STM32 Communication - UART)
# 设备: STM32F103 (小脑)
# 位置: 左侧剩余引脚
# ------------------------------------------------------------------------------
# TX_to_STM32       : GPIO 17  (发送 - 接 STM32 PA10)
# RX_from_STM32     : GPIO 18  (接收 - 接 STM32 PA9)

# ------------------------------------------------------------------------------
# [4] 听觉系统 (Audio - I2S Interface)
# 设备: INMP441 (Mic) + MAX98357A (Amp)
# 总线: I2S0
# 位置: 右侧下方连续引脚
# ------------------------------------------------------------------------------
# Amp_DOUT          : GPIO 39  (功放数据 - 接 MAX98357A DIN)
# Mic_DIN           : GPIO 40  (麦克风数据 - 接 INMP441 SD)
# I2S_LRCK          : GPIO 41  (左右声道时钟 - 麦克/功放共用 WS)
# I2S_BCLK          : GPIO 42  (位时钟 - 麦克/功放共用 SCK)

# ------------------------------------------------------------------------------
# [5] 触觉系统 (Touch - SPI Interface)
# 设备: XPT2046 (触摸屏)
# 总线: SPI (独立)
# 位置: 右侧上方零散引脚 + 左侧 GPIO 46
# ------------------------------------------------------------------------------
# Touch_CLK         : GPIO 1   (SPI 时钟)
# Touch_MOSI        : GPIO 2   (SPI 数据写)
# Touch_MISO        : GPIO 47  (SPI 数据读)
# Touch_CS          : GPIO 38  (片选 - 选中触摸芯片)
# Touch_IRQ         : GPIO 46  (触摸中断 - 按下触发低电平)

# ------------------------------------------------------------------------------
# [6] 验证外设 (Verification)
# ------------------------------------------------------------------------------
# Test_Button       : GPIO 21  (外接杜邦线按钮，用于移植测试)
#                      -> 连接方式: 一端接 GPIO 21，一端接 GND (开启内部上拉)

# ==============================================================================
# 硬件接线总结
# ------------------------------------------------------------------------------
# 左侧排针: 几乎全插满，连接到屏幕排线 (LCD数据总线 D0-D7 + 控制信号) 和 STM32
# 右侧排针: 下半部分插音频模块 (I2S)，上半部分插触摸线的剩余几根 (SPI)
# 电源要求: 屏幕背光和音频功放建议直接取 5V (开发板上的 5V 引脚)
# ==============================================================================
